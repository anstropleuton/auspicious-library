94template<typename Container>
95concept cu_compatible = requires {
96 { std::begin(std::declval<Container>()) } -> std::contiguous_iterator;
97 { std::end(std::declval<Container>()) } -> std::contiguous_iterator;
98 { std::declval<typename Container::value_type>() };
99 { Container(std::begin(std::declval<Container>()),
100 std::end(std::declval<Container>())) };
101 { Container({ std::declval<typename Container::value_type>() }) };
102};

278template<cu_compatible Container>
279[[nodiscard]] inline constexpr auto repeat(
280 const Container &container,
281 long double n
282)
283{
284 long double i_part = 0;
285 long double f_part = std::modf(n, &i_part);
286 std::size_t regular_repeat = i_part;
287 std::size_t sub_size = std::floor(f_part * container.size());
288 Container sub = subordinate(container, 0, sub_size);
289
290 return combine(repeat(container, regular_repeat), sub);
291}

319template<cu_compatible Container>
320[[nodiscard]] inline constexpr auto split_occ(
321 const Container &container,
322 const Container &values
323)
324{
325 // Couldn't find standard library to do this heavy job, so...
326 result_container_nested<Container> result;
327 auto it = container.begin();
328
329 while (it != container.end())
330 {
331 auto next_it = std::find_first_of(it, container.end(), values.begin(),
332 values.end());
333 result.emplace_back(result_container<Container>(it, next_it));
334 it = next_it;
335 if (it != container.end()) ++it;
336 }
337
338 return result;
339}
