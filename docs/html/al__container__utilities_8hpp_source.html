96concept cu_compatible = requires {
97 { std::begin(std::declval<Container>()) } -> std::contiguous_iterator;
98 { std::end(std::declval<Container>()) } -> std::contiguous_iterator;

279template<cu_compatible Container>
280[[nodiscard]] inline constexpr auto repeat(
281 const Container &container,
282 long double n
283)
284{
285 long double i_part = 0;
286 long double f_part = std::modf(n, &i_part);
287 std::size_t regular_repeat = i_part;
288 std::size_t sub_size = std::floor(f_part * container.size());
289 Container sub = subordinate(container, 0, sub_size);
290
291 return combine(repeat(container, regular_repeat), sub);
292}

320template<cu_compatible Container>
321[[nodiscard]] inline constexpr auto split_occ(
322 const Container &container,
323 const Container &values
324)
325{
326 // Couldn't find standard library to do this heavy job, so...
327 result_container_nested<Container> result;
328 auto it = container.begin();
329
330 while (it != container.end())
331 {
332 auto next_it = std::find_first_of(it, container.end(), values.begin(),
333 values.end());
334 result.emplace_back(result_container<Container>(it, next_it));
335 it = next_it;
336 if (it != container.end()) ++it;
337 }
338
339 return result;
340}
