319template<cu_compatible Container>
320[[nodiscard]] inline constexpr auto repeat(
321 const Container &container,
322 long double n
323)
324{
325 long double i_part = 0;
326 long double f_part = std::modf(n, &i_part);
327 std::size_t regular_repeat = i_part;
328 std::size_t sub_size = std::floor(f_part * container.size());
329 Container sub = subordinate(container, 0, sub_size);
330
331 return combine(repeat(container, regular_repeat), sub);
332}

360template<cu_compatible Container>
361[[nodiscard]] inline constexpr auto split_occ(
362 const Container &container,
363 const Container &values
364)
365{
366 // Couldn't find standard library to do this heavy job, so...
367 result_container_nested<Container> result;
368 auto it = container.begin();
369
370 while (it != container.end())
371 {
372 auto next_it = std::find_first_of(it, container.end(), values.begin(),
373 values.end());
374 result.emplace_back(result_container<Container>(it, next_it));
375 it = next_it;
376 if (it != container.end()) ++it;
377 }
378
379 return result;
