80template<typename Container>
81concept cu_compatible = requires {
82 { std::begin(std::declval<Container>()) } -> std::contiguous_iterator;
83 { std::end(std::declval<Container>()) } -> std::contiguous_iterator;
84 { std::declval<typename Container::value_type>() };
85 { Container(std::begin(std::declval<Container>()),
86 std::end(std::declval<Container>())) };
87 { Container({ std::declval<typename Container::value_type>() }) };
88};

263template<cu_compatible Container>
264[[nodiscard]] inline constexpr auto repeat(
265 const Container &container,
266 long double n
267)
268{
269 long double i_part = 0;
270 long double f_part = std::modf(n, &i_part);
271 std::size_t regular_repeat = i_part;
272 std::size_t sub_size = std::floor(f_part * container.size());
273 Container sub = subordinate(container, 0, sub_size);
274
275 return combine(repeat(container, regular_repeat), sub);
276}

304template<cu_compatible Container>
305[[nodiscard]] inline constexpr auto split_occ(
306 const Container &container,
307 const Container &values
308)
309{
310 // Couldn't find standard library to do this heavy job, so...
311 result_container_nested<Container> result;
312 auto it = container.begin();
313
314 while (it != container.end())
315 {
316 auto next_it = std::find_first_of(it, container.end(), values.begin(),
317 values.end());
318 result.emplace_back(result_container<Container>(it, next_it));
319 it = next_it;
320 if (it != container.end()) ++it;
321 }
322
323 return result;
324}
