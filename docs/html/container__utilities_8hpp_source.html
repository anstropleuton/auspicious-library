76template<typename Container>
77concept cu_compatible = requires {
78 { std::begin(std::declval<Container>()) } -> std::contiguous_iterator;
79 { std::end(std::declval<Container>()) } -> std::contiguous_iterator;
80 { std::declval<typename Container::value_type>() };
81 { Container(std::begin(std::declval<Container>()),
82 std::end(std::declval<Container>())) };
83 { Container({ std::declval<typename Container::value_type>() }) };
84};

258template<cu_compatible Container>
259[[nodiscard]] inline constexpr auto repeat(
260 const Container &container,
261 long double n
262)
263{
264 long double i_part = 0;
265 long double f_part = std::modf(n, &i_part);
266 std::size_t regular_repeat = i_part;
267 std::size_t sub_size = std::floor(f_part * container.size());
268 Container sub = subordinate(container, 0, sub_size);
269
270 return combine(repeat(container, regular_repeat), sub);
271}

299template<cu_compatible Container>
300[[nodiscard]] inline constexpr auto split_occ(
301 const Container &container,
302 const Container &values
303)
304{
305 // Couldn't find standard library to do this heavy job, so...
306 std::vector<result_container<Container>> result;
307 auto it = container.begin();
308
309 while (it != container.end())
310 {
311 auto next_it = std::find_first_of(it, container.end(), values.begin(),
312 values.end());
313 result.emplace_back(result_container<Container>(it, next_it));
314 it = next_it;
315 if (it != container.end()) ++it;
316 }
317
318 return result;
319}
