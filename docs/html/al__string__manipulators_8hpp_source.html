109template<cu::cu_compatible Container, typename Converter>
110requires(!std::is_same_v<Converter, std::string>)
111[[nodiscard]] inline constexpr auto to_string(
112 const Container &container,
113 Converter converter,
114 std::string_view separator = ", ",
115 std::string_view prefix = "",
116 std::string_view suffix = ""
117)
118{
119 auto transformer = [&](const cu::value_type<Container> &element)
120 {
122 return std::string(prefix) + converter(element) + std::string(suffix);
123 };
124
125 return stdv::transform(container, transformer)
126 | stdv::join_with(std::string(separator))
127 | stdr::to<std::string>();
128}

244[[nodiscard]] inline constexpr auto word_wrap(
245 std::string_view string,
246 std::size_t width,
247 bool force = false,
248 std::string_view delims = " \t\r\n\f\v\b"
249)
250{
251 result_string_nested lines = {};
252
253 // Functions expect inclusive width, and also works as a measure to have at
254 // least one as width (as a side effect)
255 width++;
256
257 while (string.size() > width)
258 {
259 // First non-delim character before or at width
260 auto pos = string.substr(0, width).find_last_of(delims);
261 if (pos == std::string_view::npos)
262 {
263 // Split without consuming character
264 if (force)
265 {
266 pos = string.substr(0, width).size() - 1;
267 auto line = string.substr(0, pos);
268 string = string.substr(pos);
269 lines.emplace_back(std::string(line));
270 continue;
271 }
272
273 // If not, first non-delim character after width
274 pos = string.substr(width + 1).find_first_of(delims);
275
276 // If still not, eof reached? Prevent overflow
277 if (pos == std::string_view::npos) pos--;
278 else pos += width + 1;
279 }
280
281 auto line = string.substr(0, pos);
282 string = string.substr(pos + 1);
283 lines.emplace_back(std::string(line));
284 }
285
286 if (!string.empty())
287 {
288 lines.emplace_back(std::string(string));
289 }
290
291 return lines;
292}

302[[nodiscard]] inline constexpr auto trim_left(
303 std::string_view string,
304 std::string_view delims = " \t\r\n\f\v\b"
305)
306{
307 auto pos = string.find_first_not_of(delims);
308 if (pos == std::string_view::npos)
309 {
310 return string;
