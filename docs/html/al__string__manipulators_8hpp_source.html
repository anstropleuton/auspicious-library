102template<cu::cu_compatible Container, typename Converter>
103requires(!std::is_same_v<Converter, std::string>)
104[[nodiscard]] inline constexpr auto to_string(
105 const Container &container,
106 Converter converter,
107 std::string_view separator = ", ",
108 std::string_view prefix = "",
109 std::string_view suffix = ""
110)
111{
112 auto transformer = [&](const cu::value_type<Container> &element)
113 {
115 return std::string(prefix) + converter(element) + std::string(suffix);
116 };
117
118 return stdv::transform(container, transformer)
119 | stdv::join_with(std::string(separator))
120 | stdr::to<std::string>();
121}

237[[nodiscard]] inline constexpr auto word_wrap(
238 std::string_view string,
239 std::size_t width,
240 bool force = false,
241 std::string_view delims = " \t\r\n\f\v\b"
242)
243{
244 std::vector<std::string> lines = {};
245
246 // Functions expect inclusive width, and also works as a measure to have at
247 // least one as width (as a side effect)
248 width++;
249
250 while (string.size() > width)
251 {
252 // First non-delim character before or at width
253 auto pos = string.substr(0, width).find_last_of(delims);
254 if (pos == std::string_view::npos)
255 {
256 // Split without consuming character
257 if (force)
258 {
259 pos = string.substr(0, width).size() - 1;
260 auto line = string.substr(0, pos);
261 string = string.substr(pos);
262 lines.emplace_back(std::string(line));
263 continue;
264 }
265
266 // If not, first non-delim character after width
267 pos = string.substr(width + 1).find_first_of(delims);
268
269 // If still not, eof reached? Prevent overflow
270 if (pos == std::string_view::npos) pos--;
271 else pos += width + 1;
272 }
273
274 auto line = string.substr(0, pos);
275 string = string.substr(pos + 1);
276 lines.emplace_back(std::string(line));
277 }
278
279 if (!string.empty())
280 {
281 lines.emplace_back(std::string(string));
282 }
283
284 return lines;
285}

298)
299{
