101requires(!std::is_same_v<Converter, std::string>)
102[[nodiscard]] inline constexpr auto to_string(
103 const Container &container,
104 Converter converter,
105 std::string_view separator = ", ",
106 std::string_view prefix = "",
107 std::string_view suffix = ""
108)
109{
110 auto transformer = [&](const cu::value_type<Container> &element)
111 {
113 return std::string(prefix) + converter(element) + std::string(suffix);
114 };
115
116 return std::views::transform(container, transformer)
117 | std::views::join_with(std::string(separator))
118 | std::ranges::to<std::string>();
119}

235[[nodiscard]] inline constexpr auto word_wrap(
236 std::string_view string,
237 std::size_t width,
238 bool force = false,
239 std::string_view delims = " \t\r\n\f\v\b"
240)
241{
242 result_string_nested lines = {};
243
244 // Functions expect inclusive width, and also works as a measure to have at
245 // least one as width (as a side effect)
246 width++;
247
248 while (string.size() > width)
249 {
250 // First non-delim character before or at width
251 auto pos = string.substr(0, width).find_last_of(delims);
252 if (pos == std::string_view::npos)
253 {
254 // Split without consuming character
255 if (force)
256 {
257 pos = string.substr(0, width).size() - 1;
258 auto line = string.substr(0, pos);
259 string = string.substr(pos);
260 lines.emplace_back(std::string(line));
261 continue;
262 }
263
264 // If not, first non-delim character after width
265 pos = string.substr(width + 1).find_first_of(delims);
266
267 // If still not, eof reached? Prevent overflow
268 if (pos == std::string_view::npos) pos--;
269 else pos += width + 1;
270 }
271
272 auto line = string.substr(0, pos);
273 string = string.substr(pos + 1);
274 lines.emplace_back(std::string(line));
275 }
276
277 if (!string.empty())
278 {
279 lines.emplace_back(std::string(string));
280 }
281
282 return lines;
283}
