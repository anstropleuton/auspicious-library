101template<cu::cu_compatible Container, typename Converter>
102requires(!std::is_same_v<Converter, std::string>)
103[[nodiscard]] inline constexpr auto to_string(
104 const Container &container,
105 Converter converter,
106 std::string_view separator = ", ",
107 std::string_view prefix = "",
108 std::string_view suffix = ""
109)
110{
111 auto transformer = [&](const cu::value_type<Container> &element)
112 {
114 return std::string(prefix) + converter(element) + std::string(suffix);
115 };
116
117 return stdv::transform(container, transformer)
118 | stdv::join_with(std::string(separator))
119 | stdr::to<std::string>();
120}

236[[nodiscard]] inline constexpr auto word_wrap(
237 std::string_view string,
238 std::size_t width,
239 bool force = false,
240 std::string_view delims = " \t\r\n\f\v\b"
241)
242{
243 std::vector<std::string> lines = {};
244
245 // Functions expect inclusive width, and also works as a measure to have at
246 // least one as width (as a side effect)
247 width++;
248
249 while (string.size() > width)
250 {
251 // First non-delim character before or at width
252 auto pos = string.substr(0, width).find_last_of(delims);
253 if (pos == std::string_view::npos)
254 {
255 // Split without consuming character
256 if (force)
257 {
258 pos = string.substr(0, width).size() - 1;
259 auto line = string.substr(0, pos);
260 string = string.substr(pos);
261 lines.emplace_back(std::string(line));
262 continue;
263 }
264
265 // If not, first non-delim character after width
266 pos = string.substr(width + 1).find_first_of(delims);
267
268 // If still not, eof reached? Prevent overflow
269 if (pos == std::string_view::npos) pos--;
270 else pos += width + 1;
271 }
272
273 auto line = string.substr(0, pos);
274 string = string.substr(pos + 1);
275 lines.emplace_back(std::string(line));
276 }
277
278 if (!string.empty())
279 {
280 lines.emplace_back(std::string(string));
281 }
282
283 return lines;
284}

297)
