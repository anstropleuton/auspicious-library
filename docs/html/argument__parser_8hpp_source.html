236[[nodiscard]] inline constexpr auto to_string(argument_type arg_type)
237{
238 switch (arg_type)
239 {
240 case argument_type::unknown: return "unknown"s;
241 case argument_type::empty: return "empty"s;
242 case argument_type::short_option: return "short_option"s;
243 case argument_type::long_option: return "long_option"s;
244 case argument_type::microsoft_switch: return "microsoft_switch"s;
245 case argument_type::single_hyphen: return "single_hyphen"s;
246 case argument_type::double_hyphen: return "double_hyphen"s;
247 case argument_type::regular_argument: return "regular_argument"s;
248 }
249}
250
257[[nodiscard]] inline constexpr auto get_argument_type(std::string_view argument)
258{
259 // I should probably refactor this to make it more readable rather than
260 // making it never-nester-friendly
261 if (argument.size() == 0) return argument_type::regular_argument;
262 if (argument[0] == '/') return argument_type::microsoft_switch;
263 if (argument[0] != '-') return argument_type::regular_argument;
264 if (argument.size() == 1) return argument_type::single_hyphen;
265 if (argument[1] == '-')
266 {
267 if (argument.size() == 2) return argument_type::double_hyphen;
268 return argument_type::long_option;
269 }
270 return argument_type::short_option;
271}

304[[nodiscard]] inline constexpr auto is_parameters_variadic(
305 const std::vector<std::string> &parameters
306)
307{
308 if (parameters.empty())
309 return variadicity::not_variadic;
310 if (parameters.back() == "...")
311 return variadicity::zero_or_more;
312 if (parameters.back().ends_with("..."))
313 return variadicity::one_or_more;
314 return variadicity::not_variadic;
315}
