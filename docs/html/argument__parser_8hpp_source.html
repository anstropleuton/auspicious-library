261[[nodiscard]] inline constexpr auto get_argument_type(std::string_view argument)
262{
263 // I should probably refactor this to make it more readable rather than
264 // making it never-nester-friendly
265 if (argument.size() == 0) return argument_type::regular_argument;
266 if (argument[0] == '/') return argument_type::microsoft_switch;
267 if (argument[0] != '-') return argument_type::regular_argument;
268 if (argument.size() == 1) return argument_type::single_hyphen;
269 if (argument[1] == '-')
270 {
271 if (argument.size() == 2) return argument_type::double_hyphen;
272 return argument_type::long_option;
273 }
274 return argument_type::short_option;
275}

308[[nodiscard]] inline constexpr auto is_parameters_variadic(
309 const std::vector<std::string> &parameters
310)
311{
312 if (parameters.empty())
313 return variadicity::not_variadic;
314 if (parameters.back() == "...")
315 return variadicity::zero_or_more;
316 if (parameters.back().ends_with("..."))
317 return variadicity::one_or_more;
318 return variadicity::not_variadic;
