310[[nodiscard]] inline constexpr auto get_argument_type(std::string_view argument)
311{
312 if (argument.size() == 0) return argument_type::empty;
313
314 if (argument.starts_with("--"))
315 {
316 if (argument.size() > 2) return argument_type::long_option;
317 else return argument_type::double_hyphen;
318 }
319
320 if (argument.starts_with("-"))
321 {
322 if (argument.size() > 1) return argument_type::short_option;
323 else return argument_type::single_hyphen;
324 }
325
326 if (argument.starts_with("/"))
327 {
328 return argument_type::microsoft_switch;
329 }
330
331 return argument_type::regular_argument;
332}
