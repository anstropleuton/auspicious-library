315[[nodiscard]] inline constexpr auto get_argument_type(std::string_view argument)
316{
317 if (argument.size() == 0) return argument_type::empty;
318
319 if (argument.starts_with("--"))
320 {
321 if (argument.size() > 2) return argument_type::long_option;
322 else return argument_type::double_hyphen;
323 }
324
325 if (argument.starts_with("-"))
326 {
327 if (argument.size() > 1) return argument_type::short_option;
328 else return argument_type::single_hyphen;
329 }
330
331 if (argument.starts_with("/"))
332 {
333 return argument_type::microsoft_switch;
334 }
335
336 return argument_type::regular_argument;
337}
