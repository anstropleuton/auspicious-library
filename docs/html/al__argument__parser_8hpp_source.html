297[[nodiscard]] inline constexpr auto get_argument_type(std::string_view argument)
298{
299 if (argument.size() == 0) return argument_type::empty;
300
301 if (argument.starts_with("--"))
302 {
303 if (argument.size() > 2) return argument_type::long_option;
304 else return argument_type::double_hyphen;
305 }
306
307 if (argument.starts_with("-"))
308 {
309 if (argument.size() > 1) return argument_type::short_option;
310 else return argument_type::single_hyphen;
311 }
312
313 if (argument.starts_with("/"))
314 {
315 return argument_type::microsoft_switch;
316 }
317
318 return argument_type::regular_argument;
319}

692struct styled_padding {

697 styled_text first;

702 styled_text mid;
703
707 styled_text last;
708
716 std::size_t width;
717
724 [[nodiscard]] inline constexpr auto str(std::size_t subtract = 0) const
725 {
726 using namespace std::string_literals;
727 if (subtract > width)
728 {
729 return ""s;
730 }
731
732 std::size_t actual_size = width - subtract;
733 if (actual_size == 0)
734 {
735 return ""s;
736 }
737 if (actual_size == 1)
738 {
739 return mid.style(mid.value);
740 }
741 if (actual_size == 2)
742 {
743 return first.style(first.value)
744 + last.style(last.value);
745 }
746 return first.style(first.value)
747 + mid.style(sm::repeat(mid.value, actual_size))
748 + last.style(last.value);
749 }
750
757 [[nodiscard]] inline constexpr auto size(std::size_t subtract = 0) const
758 {
759 if (subtract > width)
760 {
761 return 0zu;
762 }
763
764 std::size_t actual_size = width - subtract;
765 if (actual_size == 0)
766 {
767 return 0zu;
768 }
769 if (actual_size == 1)
770 {
771 return mid.value.size();
772 }
773 if (actual_size == 2)
774 {
775 return first.value.size()
776 + last.value.size();
777 }
778 return first.value.size()
779 + mid.value.size() * actual_size
780 + last.value.size();
781 }
782
789 [[nodiscard]] inline constexpr auto m_str(std::size_t subtract = 0) const
790 {
791 return measured_string(str(subtract), size(subtract));
792 }
