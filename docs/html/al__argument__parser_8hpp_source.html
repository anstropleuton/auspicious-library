264[[nodiscard]] inline constexpr auto get_argument_type(std::string_view argument)
265{
266 if (argument.size() == 0) return argument_type::empty;
267
268 if (argument.starts_with("--"))
269 {
270 if (argument.size() > 2) return argument_type::long_option;
271 else return argument_type::double_hyphen;
272 }
273
274 if (argument.starts_with("-"))
275 {
276 if (argument.size() > 2) return argument_type::short_option;
277 else return argument_type::single_hyphen;
278 }
279
280 if (argument.starts_with("/"))
281 {
282 return argument_type::microsoft_switch;
283 }
284
285 return argument_type::regular_argument;
286}
