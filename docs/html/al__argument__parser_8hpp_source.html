316[[nodiscard]] inline constexpr auto get_argument_type(std::string_view argument)
317{
318 if (argument.size() == 0) return argument_type::empty;
319
320 if (argument.starts_with("--"))
321 {
322 if (argument.size() > 2) return argument_type::long_option;
323 else return argument_type::double_hyphen;
324 }
325
326 if (argument.starts_with("-"))
327 {
328 if (argument.size() > 1) return argument_type::short_option;
329 else return argument_type::single_hyphen;
330 }
331
332 if (argument.starts_with("/"))
333 {
334 return argument_type::microsoft_switch;
335 }
336
337 return argument_type::regular_argument;
338}

709struct styled_padding {
710
714 styled_text first;
715
719 styled_text mid;
720
724 styled_text last;
725
733 std::size_t width;
734
741 [[nodiscard]] inline constexpr auto str(std::size_t subtract = 0) const
742 {
743 if (subtract > width)
744 {
745 return ""s;
746 }
747
748 std::size_t actual_size = width - subtract;
749 if (actual_size == 0)
750 {
751 return ""s;
752 }
753 if (actual_size == 1)
754 {
755 return mid.style(mid.value);
756 }
757 if (actual_size == 2)
758 {
759 return first.style(first.value)
760 + last.style(last.value);
761 }
762 return first.style(first.value)
763 + mid.style(sm::repeat(mid.value, actual_size))
764 + last.style(last.value);
765 }
766
773 [[nodiscard]] inline constexpr auto size(std::size_t subtract = 0) const
774 {
775 if (subtract > width)
776 {
777 return 0zu;
778 }
779
780 std::size_t actual_size = width - subtract;
781 if (actual_size == 0)
782 {
783 return 0zu;
784 }
785 if (actual_size == 1)
786 {
787 return mid.value.size();
788 }
789 if (actual_size == 2)
790 {
791 return first.value.size()
792 + last.value.size();
793 }
794 return first.value.size()
795 + mid.value.size() * actual_size
796 + last.value.size();
797 }
798
805 [[nodiscard]] inline constexpr auto m_str(std::size_t subtract = 0) const
806 {
807 return measured_string(str(subtract), size(subtract));
808 }
809};
