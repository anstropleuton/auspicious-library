311[[nodiscard]] inline constexpr auto get_argument_type(std::string_view argument)
312{
313 if (argument.size() == 0) return argument_type::empty;
314
315 if (argument.starts_with("--"))
316 {
317 if (argument.size() > 2) return argument_type::long_option;
318 else return argument_type::double_hyphen;
319 }
320
321 if (argument.starts_with("-"))
322 {
323 if (argument.size() > 1) return argument_type::short_option;
324 else return argument_type::single_hyphen;
325 }
326
327 if (argument.starts_with("/"))
328 {
329 return argument_type::microsoft_switch;
330 }
331
332 return argument_type::regular_argument;
333}
