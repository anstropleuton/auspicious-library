82template<cu::cu_compatible Container, typename Converter>
83[[nodiscard]] inline constexpr auto to_string(
84 const Container &container,
85 Converter converter,
86 std::string_view separator = ", ",
87 std::string_view prefix = "",
88 std::string_view suffix = ""
89)
90{
91 auto transformer = [&](const cu::value_type<Container> &element)
92 {
94 return std::string(prefix) + converter(element) + std::string(suffix);
95 };
96
97 return stdv::transform(container, transformer)
98 | stdv::join_with(std::string(separator))
99 | stdr::to<std::string>();
100}

216[[nodiscard]] inline constexpr auto word_wrap(
217 std::string_view string,
218 std::size_t width,
219 bool force = false,
220 std::string_view delims = " \t\r\n\f\v\b"
221)
222{
223 std::vector<std::string> lines = {};
224
225 // Functions expect inclusive width, and also works as a measure to have at
226 // least one as width (as a side effect)
227 width++;
228
229 while (string.size() > width)
230 {
231 // First non-delim character before or at width
232 auto pos = string.substr(0, width).find_last_of(delims);
233 if (pos == std::string_view::npos)
234 {
235 // Split without consuming character
236 if (force)
237 {
238 pos = string.substr(0, width).size() - 1;
239 auto line = string.substr(0, pos);
240 string = string.substr(pos);
241 lines.emplace_back(std::string(line));
242 continue;
243 }
244
245 // If not, first non-delim character after width
246 pos = string.substr(width + 1).find_first_of(delims);
247
248 // If still not, eof reached? Prevent overflow
249 if (pos == std::string_view::npos) pos--;
250 else pos += width + 1;
251 }
252
253 auto line = string.substr(0, pos);
254 string = string.substr(pos + 1);
255 lines.emplace_back(std::string(line));
256 }
257
258 if (!string.empty())
259 {
260 lines.emplace_back(std::string(string));
261 }
262
263 return lines;
264}

274[[nodiscard]] inline constexpr auto trim_left(
275 std::string_view string,
276 std::string_view delims = " \t\r\n\f\v\b"
277)
278{
279 auto pos = string.find_first_not_of(delims);
280 if (pos == std::string_view::npos)
281 {
282 return string;
283 }
284 return string.substr(pos);
285}

295[[nodiscard]] inline constexpr auto trim_right(
296 std::string_view string,
297 std::string_view delims = " \t\r\n\f\v\b"
298)
299{
300 auto pos = string.find_last_not_of(delims);
301 if (pos == std::string_view::npos)
302 {
303 return string;
304 }
305 return string.substr(0, pos + 1);
