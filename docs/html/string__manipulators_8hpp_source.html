86template<cu::cu_compatible Container, typename Converter>
87[[nodiscard]] inline constexpr auto to_string(
88 const Container &container,
89 Converter converter,
90 std::string_view separator = ", ",
91 std::string_view prefix = "",
92 std::string_view suffix = ""
93)
94{
95 auto transformer = [&](const cu::value_type<Container> &element)
96 {
98 return std::string(prefix) + converter(element) + std::string(suffix);
99 };
100
101 return stdv::transform(container, transformer)
102 | stdv::join_with(std::string(separator))
103 | stdr::to<std::string>();
104}

220[[nodiscard]] inline constexpr auto word_wrap(
221 std::string_view string,
222 std::size_t width,
223 bool force = false,
224 std::string_view delims = " \t\r\n\f\v\b"
225)
226{
227 std::vector<std::string> lines = {};
228
229 // Functions expect inclusive width, and also works as a measure to have at
230 // least one as width (as a side effect)
231 width++;
232
233 while (string.size() > width)
234 {
235 // First non-delim character before or at width
236 auto pos = string.substr(0, width).find_last_of(delims);
237 if (pos == std::string_view::npos)
238 {
239 // Split without consuming character
240 if (force)
241 {
242 pos = string.substr(0, width).size() - 1;
243 auto line = string.substr(0, pos);
244 string = string.substr(pos);
245 lines.emplace_back(std::string(line));
246 continue;
247 }
248
249 // If not, first non-delim character after width
250 pos = string.substr(width + 1).find_first_of(delims);
251
252 // If still not, eof reached? Prevent overflow
253 if (pos == std::string_view::npos) pos--;
254 else pos += width + 1;
255 }
256
257 auto line = string.substr(0, pos);
258 string = string.substr(pos + 1);
259 lines.emplace_back(std::string(line));
260 }
261
262 if (!string.empty())
263 {
264 lines.emplace_back(std::string(string));
265 }
266
267 return lines;
268}
