135struct test_suite {
136
140 std::vector<const test *> tests;
141
145 std::function<void (const test *)> pre_run;
146
150 std::function<void (const test *, std::size_t)> post_run;
151
157 std::function<bool (const test *, std::size_t)> run_failed;
158
163 [[nodiscard]] inline constexpr auto run()
164 {
165 std::vector<std::pair<const test *, std::size_t>> failed_tests = {};
166 for (auto &test : tests)
167 {
168 if (pre_run) pre_run(test);
169 std::size_t errors = test->run();
170 bool end_test = false;
171
172 if (errors != 0)
173 {
174 failed_tests.emplace_back(test, errors);
175 if (run_failed && run_failed(test, errors))
176 {
177 end_test = true;
178 }
179 }
180
181 if (post_run) post_run(test, errors);
182 if (end_test) break;
183 }
184 return failed_tests;
185 };
186};
